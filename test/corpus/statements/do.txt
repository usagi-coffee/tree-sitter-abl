================================================================================
DO BLOCK
================================================================================

DO:
  x = 1.
END.

DO idx = 1 TO 3:
  x = idx.
END.

DO WHILE idx < 3 ON ERROR UNDO, THROW:
  idx = idx + 1.
END.

DO ON ERROR UNDO, LEAVE:
  idx = 1.
END.

DO ON QUIT, LEAVE:
  idx = 2.
END.

DO TRANSACTION:
END.

DO TRANSACTION WHILE something:get-next():
  something:buffer-copy(something).
END.

DO WHILE TRUE ON ENDKEY UNDO, RETRY:
  MESSAGE "hi".
END.
